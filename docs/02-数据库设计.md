# AegisCDN 数据库设计 (PostgreSQL)

版本 v1.0 · 2026-05-21 · 引擎 PostgreSQL 14+ · 命名 snake_case · 主键 `bigserial` / 部分 `uuid`

> 约定:
> - 多租户隔离字段统一为 `tenant_id`(平台级表如 plan/node/announcement 无此字段)。
> - 金额用 `numeric(14,4)`(元),流量字节用 `bigint`,时间统一 `timestamptz`(UTC)。
> - 软删除用 `deleted_at timestamptz null`;审计字段 `created_at` / `updated_at`。
> - 实际以 Prisma schema 为准,本文给出等价 DDL 以便评审。

## 0. 表清单总览

| 域 | 表 |
| --- | --- |
| 租户与账号 | `tenant` `user` `role` `permission` `role_permission` `api_key` |
| 计费 | `plan` `subscription` `order` `payment` `wallet` `wallet_txn` `coupon` `coupon_redemption` |
| 域名与接入 | `domain` `origin` `certificate` `cache_rule` |
| 安全配置 | `cc_policy` `waf_policy` `waf_rule` `acl_rule` |
| 配置下发 | `config_version` `deploy_task` |
| 边缘节点 | `node` `node_group` |
| 统计与日志 | `traffic_stat` `attack_event` `purge_task` |
| 运营 | `announcement` `ticket` `ticket_message` `audit_log` `global_block` `notification` |

ER 关系(简):`tenant 1—N user`,`tenant 1—N domain`,`domain 1—N origin/cache_rule/acl_rule/config_version`,
`domain 1—1 cc_policy/waf_policy/certificate`,`tenant 1—1 subscription —N order`,`node N—1 node_group`。

## 1. 租户与账号

```sql
-- 租户(企业客户)
CREATE TABLE tenant (
  id            bigserial PRIMARY KEY,
  name          varchar(128) NOT NULL,           -- 企业/客户名称
  status        varchar(16)  NOT NULL DEFAULT 'active', -- active|suspended|closed
  company_info  jsonb,                            -- KYC:营业执照/联系人等
  balance       numeric(14,4) NOT NULL DEFAULT 0, -- 钱包余额(冗余,以 wallet 为准)
  created_at    timestamptz NOT NULL DEFAULT now(),
  updated_at    timestamptz NOT NULL DEFAULT now(),
  deleted_at    timestamptz
);

-- 账号(租户成员;平台角色账号 tenant_id 可为空)
CREATE TABLE "user" (
  id            bigserial PRIMARY KEY,
  tenant_id     bigint REFERENCES tenant(id),     -- 平台 operator/admin 可为 NULL
  email         varchar(255) NOT NULL,
  password_hash varchar(255) NOT NULL,
  name          varchar(128),
  phone         varchar(32),
  role          varchar(16)  NOT NULL DEFAULT 'user', -- user|member|operator|admin
  is_active     boolean NOT NULL DEFAULT true,
  email_verified_at timestamptz,
  last_login_at timestamptz,
  totp_secret   varchar(64),                      -- 2FA(可选)
  created_at    timestamptz NOT NULL DEFAULT now(),
  updated_at    timestamptz NOT NULL DEFAULT now(),
  deleted_at    timestamptz
);
CREATE UNIQUE INDEX uq_user_email ON "user"(lower(email)) WHERE deleted_at IS NULL;
CREATE INDEX idx_user_tenant ON "user"(tenant_id);

-- RBAC(细粒度;role 字段做粗粒度,permission 做细粒度叠加)
CREATE TABLE role (
  id     bigserial PRIMARY KEY,
  code   varchar(32) UNIQUE NOT NULL,             -- user|member|operator|admin|custom_*
  name   varchar(64) NOT NULL,
  scope  varchar(16) NOT NULL                     -- tenant|platform
);
CREATE TABLE permission (
  id     bigserial PRIMARY KEY,
  code   varchar(64) UNIQUE NOT NULL,             -- domain.read / waf.write / admin.user.ban ...
  name   varchar(128) NOT NULL,
  module varchar(32) NOT NULL
);
CREATE TABLE role_permission (
  role_id       bigint REFERENCES role(id) ON DELETE CASCADE,
  permission_id bigint REFERENCES permission(id) ON DELETE CASCADE,
  PRIMARY KEY (role_id, permission_id)
);

-- API 密钥(供客户调 OpenAPI)
CREATE TABLE api_key (
  id          bigserial PRIMARY KEY,
  tenant_id   bigint NOT NULL REFERENCES tenant(id),
  name        varchar(64) NOT NULL,
  key_id      varchar(32) UNIQUE NOT NULL,        -- 公开标识
  secret_hash varchar(255) NOT NULL,              -- 仅存哈希
  scopes      jsonb NOT NULL DEFAULT '[]',
  last_used_at timestamptz,
  expires_at  timestamptz,
  revoked_at  timestamptz,
  created_at  timestamptz NOT NULL DEFAULT now()
);
CREATE INDEX idx_apikey_tenant ON api_key(tenant_id);
```

## 2. 计费

```sql
-- 套餐定义(平台级)
CREATE TABLE plan (
  id              bigserial PRIMARY KEY,
  code            varchar(32) UNIQUE NOT NULL,     -- starter|business|enterprise
  name            varchar(64) NOT NULL,
  tier            int NOT NULL,                     -- 排序/等级
  price_monthly   numeric(14,4) NOT NULL,
  price_yearly    numeric(14,4) NOT NULL,
  protection_gbps int,                              -- 防护容量(G);NULL=定制
  traffic_gb      bigint,                           -- 月流量额度 GB;NULL=不限/定制
  domain_limit    int NOT NULL,                     -- 可接入域名数
  request_limit   bigint,                           -- 月请求数上限;NULL=不限
  features        jsonb NOT NULL DEFAULT '{}',      -- {cc:true,waf:true,geo:true,dedicated_node:false,...}
  is_active       boolean NOT NULL DEFAULT true,
  is_custom       boolean NOT NULL DEFAULT false,   -- enterprise 定制
  created_at      timestamptz NOT NULL DEFAULT now(),
  updated_at      timestamptz NOT NULL DEFAULT now()
);

-- 租户订阅(当前生效套餐)
CREATE TABLE subscription (
  id            bigserial PRIMARY KEY,
  tenant_id     bigint NOT NULL REFERENCES tenant(id),
  plan_id       bigint NOT NULL REFERENCES plan(id),
  status        varchar(16) NOT NULL DEFAULT 'active', -- trialing|active|past_due|canceled|expired
  cycle         varchar(8)  NOT NULL DEFAULT 'monthly', -- monthly|yearly
  current_period_start timestamptz NOT NULL,
  current_period_end   timestamptz NOT NULL,
  -- 配额覆盖(企业定制时覆盖 plan 默认值)
  quota_override jsonb,
  auto_renew    boolean NOT NULL DEFAULT true,
  created_at    timestamptz NOT NULL DEFAULT now(),
  updated_at    timestamptz NOT NULL DEFAULT now()
);
CREATE INDEX idx_sub_tenant ON subscription(tenant_id);

-- 订单(购买/续费/升级/流量包)
CREATE TABLE "order" (
  id           bigserial PRIMARY KEY,
  order_no     varchar(32) UNIQUE NOT NULL,
  tenant_id    bigint NOT NULL REFERENCES tenant(id),
  type         varchar(16) NOT NULL,              -- new|renew|upgrade|addon_traffic
  plan_id      bigint REFERENCES plan(id),
  cycle        varchar(8),
  amount       numeric(14,4) NOT NULL,
  discount     numeric(14,4) NOT NULL DEFAULT 0,
  payable      numeric(14,4) NOT NULL,
  currency     varchar(8) NOT NULL DEFAULT 'CNY',
  status       varchar(16) NOT NULL DEFAULT 'pending', -- pending|paid|canceled|refunded
  meta         jsonb,
  created_at   timestamptz NOT NULL DEFAULT now(),
  paid_at      timestamptz
);
CREATE INDEX idx_order_tenant ON "order"(tenant_id);

-- 支付流水
CREATE TABLE payment (
  id           bigserial PRIMARY KEY,
  order_id     bigint NOT NULL REFERENCES "order"(id),
  method       varchar(16) NOT NULL,              -- stripe|wallet|bank_transfer
  provider_ref varchar(128),                      -- 第三方交易号
  amount       numeric(14,4) NOT NULL,
  status       varchar(16) NOT NULL,              -- pending|succeeded|failed|refunded
  raw          jsonb,
  created_at   timestamptz NOT NULL DEFAULT now()
);

-- 钱包与流水
CREATE TABLE wallet (
  tenant_id    bigint PRIMARY KEY REFERENCES tenant(id),
  balance      numeric(14,4) NOT NULL DEFAULT 0,
  updated_at   timestamptz NOT NULL DEFAULT now()
);
CREATE TABLE wallet_txn (
  id           bigserial PRIMARY KEY,
  tenant_id    bigint NOT NULL REFERENCES tenant(id),
  direction    varchar(8) NOT NULL,               -- in|out
  amount       numeric(14,4) NOT NULL,
  balance_after numeric(14,4) NOT NULL,
  reason       varchar(32) NOT NULL,              -- recharge|order|refund|adjust|overage
  ref_id       bigint,                            -- 关联 order/payment
  remark       varchar(255),
  created_at   timestamptz NOT NULL DEFAULT now()
);
CREATE INDEX idx_wallet_txn_tenant ON wallet_txn(tenant_id, created_at);

-- 优惠券(可选)
CREATE TABLE coupon (
  id          bigserial PRIMARY KEY,
  code        varchar(32) UNIQUE NOT NULL,
  type        varchar(8) NOT NULL,                -- percent|fixed
  value       numeric(14,4) NOT NULL,
  max_uses    int, used int NOT NULL DEFAULT 0,
  valid_from  timestamptz, valid_to timestamptz,
  is_active   boolean NOT NULL DEFAULT true
);
CREATE TABLE coupon_redemption (
  id         bigserial PRIMARY KEY,
  coupon_id  bigint NOT NULL REFERENCES coupon(id),
  tenant_id  bigint NOT NULL REFERENCES tenant(id),
  order_id   bigint REFERENCES "order"(id),
  created_at timestamptz NOT NULL DEFAULT now()
);
```

## 3. 域名与接入

```sql
CREATE TABLE domain (
  id            bigserial PRIMARY KEY,
  tenant_id     bigint NOT NULL REFERENCES tenant(id),
  name          varchar(253) NOT NULL,            -- 接入域名 example.com
  cname         varchar(253) NOT NULL,            -- 分配的 CNAME d-8f3a.aegis-cdn.net
  verify_token  varchar(64) NOT NULL,             -- TXT 校验值
  status        varchar(16) NOT NULL DEFAULT 'pending', -- pending|verifying|active|paused|blocked
  review_status varchar(16) NOT NULL DEFAULT 'pending', -- pending|approved|rejected(接入审核)
  protocol      varchar(8) NOT NULL DEFAULT 'https',    -- http|https|both
  force_https   boolean NOT NULL DEFAULT true,
  node_group_id bigint REFERENCES node_group(id),       -- 调度到的节点组
  active_config_version_id bigint,                       -- 当前生效配置版本
  created_at    timestamptz NOT NULL DEFAULT now(),
  updated_at    timestamptz NOT NULL DEFAULT now(),
  deleted_at    timestamptz
);
CREATE UNIQUE INDEX uq_domain_name ON domain(lower(name)) WHERE deleted_at IS NULL;
CREATE INDEX idx_domain_tenant ON domain(tenant_id);

-- 回源(支持多源/主备/负载)
CREATE TABLE origin (
  id          bigserial PRIMARY KEY,
  domain_id   bigint NOT NULL REFERENCES domain(id) ON DELETE CASCADE,
  address     varchar(253) NOT NULL,              -- ip 或 域名
  port        int NOT NULL DEFAULT 80,
  scheme      varchar(8) NOT NULL DEFAULT 'http', -- http|https
  weight      int NOT NULL DEFAULT 100,
  role        varchar(8) NOT NULL DEFAULT 'primary', -- primary|backup
  host_header varchar(253),                       -- 回源 Host 改写
  created_at  timestamptz NOT NULL DEFAULT now()
);
CREATE INDEX idx_origin_domain ON origin(domain_id);

-- HTTPS 证书
CREATE TABLE certificate (
  id          bigserial PRIMARY KEY,
  tenant_id   bigint NOT NULL REFERENCES tenant(id),
  domain_id   bigint REFERENCES domain(id) ON DELETE CASCADE,
  source      varchar(8) NOT NULL DEFAULT 'acme', -- acme|upload
  cert_pem    text,                               -- 加密存储
  key_pem     text,                               -- 加密存储(KMS/对称加密)
  issuer      varchar(128),
  sans        jsonb,
  not_before  timestamptz, not_after timestamptz,
  status      varchar(16) NOT NULL DEFAULT 'pending', -- pending|active|expiring|expired|failed
  created_at  timestamptz NOT NULL DEFAULT now()
);
CREATE INDEX idx_cert_domain ON certificate(domain_id);

-- 缓存规则
CREATE TABLE cache_rule (
  id          bigserial PRIMARY KEY,
  domain_id   bigint NOT NULL REFERENCES domain(id) ON DELETE CASCADE,
  priority    int NOT NULL DEFAULT 100,
  match_type  varchar(16) NOT NULL,               -- path|suffix|regex|status
  match_value varchar(255) NOT NULL,
  ttl_seconds int NOT NULL DEFAULT 3600,
  cache_enabled boolean NOT NULL DEFAULT true,
  ignore_query boolean NOT NULL DEFAULT false,
  created_at  timestamptz NOT NULL DEFAULT now()
);
CREATE INDEX idx_cache_domain ON cache_rule(domain_id, priority);
```

## 4. 安全配置

```sql
-- CC 防护(每域名一份)
CREATE TABLE cc_policy (
  domain_id   bigint PRIMARY KEY REFERENCES domain(id) ON DELETE CASCADE,
  enabled     boolean NOT NULL DEFAULT true,
  mode        varchar(16) NOT NULL DEFAULT 'normal', -- off|normal|attack|strict
  rate_per_min int NOT NULL DEFAULT 600,           -- 单 IP 每分钟阈值
  burst       int NOT NULL DEFAULT 50,
  action      varchar(16) NOT NULL DEFAULT 'challenge', -- log|challenge|captcha|block
  challenge_type varchar(16) DEFAULT 'js',          -- js|captcha|5s
  updated_at  timestamptz NOT NULL DEFAULT now()
);

-- WAF 策略(每域名一份)
CREATE TABLE waf_policy (
  domain_id    bigint PRIMARY KEY REFERENCES domain(id) ON DELETE CASCADE,
  enabled      boolean NOT NULL DEFAULT true,
  mode         varchar(16) NOT NULL DEFAULT 'block', -- off|observe|block
  rulesets     jsonb NOT NULL DEFAULT '["owasp_core","sqli","xss","rce","lfi"]', -- 开启的内置规则集
  bot_protection boolean NOT NULL DEFAULT true,
  updated_at   timestamptz NOT NULL DEFAULT now()
);

-- WAF 自定义规则
CREATE TABLE waf_rule (
  id          bigserial PRIMARY KEY,
  domain_id   bigint NOT NULL REFERENCES domain(id) ON DELETE CASCADE,
  priority    int NOT NULL DEFAULT 100,
  field       varchar(16) NOT NULL,               -- uri|header|arg|body|ua|referer|ip|country
  operator    varchar(16) NOT NULL,               -- eq|contains|regex|in|cidr
  value       text NOT NULL,
  action      varchar(16) NOT NULL,               -- allow|block|challenge|log
  enabled     boolean NOT NULL DEFAULT true,
  created_at  timestamptz NOT NULL DEFAULT now()
);
CREATE INDEX idx_wafrule_domain ON waf_rule(domain_id, priority);

-- ACL:IP / 地区 / UA 黑白名单(统一表)
CREATE TABLE acl_rule (
  id          bigserial PRIMARY KEY,
  domain_id   bigint NOT NULL REFERENCES domain(id) ON DELETE CASCADE,
  category    varchar(8) NOT NULL,                -- ip|geo|ua|referer
  list_type   varchar(8) NOT NULL,                -- allow|deny
  value       varchar(255) NOT NULL,              -- 1.2.3.0/24 | CN | *bot* ...
  note        varchar(255),
  enabled     boolean NOT NULL DEFAULT true,
  created_at  timestamptz NOT NULL DEFAULT now()
);
CREATE INDEX idx_acl_domain ON acl_rule(domain_id, category);
```

## 5. 配置下发

```sql
-- 域名配置版本(快照,可灰度/回滚)
CREATE TABLE config_version (
  id          bigserial PRIMARY KEY,
  domain_id   bigint NOT NULL REFERENCES domain(id) ON DELETE CASCADE,
  version     int NOT NULL,                       -- 自增版本号(按域名)
  snapshot    jsonb NOT NULL,                     -- 渲染所需的完整配置快照
  status      varchar(16) NOT NULL DEFAULT 'pending', -- pending|deploying|active|failed|rolled_back
  created_by  bigint REFERENCES "user"(id),
  created_at  timestamptz NOT NULL DEFAULT now()
);
CREATE UNIQUE INDEX uq_cfgver ON config_version(domain_id, version);

-- 下发任务(配置版本 × 节点)
CREATE TABLE deploy_task (
  id           bigserial PRIMARY KEY,
  config_version_id bigint NOT NULL REFERENCES config_version(id) ON DELETE CASCADE,
  node_id      bigint NOT NULL REFERENCES node(id),
  status       varchar(16) NOT NULL DEFAULT 'queued', -- queued|pushing|applied|failed
  attempts     int NOT NULL DEFAULT 0,
  error        text,
  applied_at   timestamptz,
  created_at   timestamptz NOT NULL DEFAULT now()
);
CREATE INDEX idx_deploy_node ON deploy_task(node_id, status);
```

## 7. 统计与日志

```sql
-- 流量/请求预聚合(分钟/小时/天三粒度,granularity 区分)
CREATE TABLE traffic_stat (
  id            bigserial PRIMARY KEY,
  tenant_id     bigint NOT NULL REFERENCES tenant(id),
  domain_id     bigint REFERENCES domain(id) ON DELETE CASCADE,
  node_id       bigint REFERENCES node(id),
  bucket        timestamptz NOT NULL,             -- 时间桶起点
  granularity   varchar(4) NOT NULL,              -- 1m|1h|1d
  requests      bigint NOT NULL DEFAULT 0,
  bytes_in      bigint NOT NULL DEFAULT 0,
  bytes_out     bigint NOT NULL DEFAULT 0,
  cache_hit     bigint NOT NULL DEFAULT 0,
  cache_miss    bigint NOT NULL DEFAULT 0,
  status_2xx    bigint NOT NULL DEFAULT 0,
  status_3xx    bigint NOT NULL DEFAULT 0,
  status_4xx    bigint NOT NULL DEFAULT 0,
  status_5xx    bigint NOT NULL DEFAULT 0,
  blocked       bigint NOT NULL DEFAULT 0         -- 被拦截请求数(防护次数)
);
CREATE UNIQUE INDEX uq_traffic ON traffic_stat(domain_id, node_id, granularity, bucket);
CREATE INDEX idx_traffic_tenant_bucket ON traffic_stat(tenant_id, bucket);
-- 生产建议:按 bucket 做分区表(pg_partman),或时序库(TimescaleDB)hypertable。

-- 攻击事件(被 WAF/CC/ACL 拦截的明细;按套餐保留期)
CREATE TABLE attack_event (
  id            bigserial PRIMARY KEY,
  tenant_id     bigint NOT NULL REFERENCES tenant(id),
  domain_id     bigint NOT NULL REFERENCES domain(id) ON DELETE CASCADE,
  node_id       bigint REFERENCES node(id),
  occurred_at   timestamptz NOT NULL,
  client_ip     inet NOT NULL,
  country       varchar(2),
  type          varchar(16) NOT NULL,             -- cc|waf|acl|bot|ratelimit
  rule          varchar(64),                      -- 命中规则标识
  action        varchar(16) NOT NULL,             -- block|challenge|log
  method        varchar(8),
  uri           varchar(1024),
  ua            varchar(512),
  raw           jsonb
);
CREATE INDEX idx_attack_domain_time ON attack_event(domain_id, occurred_at);
CREATE INDEX idx_attack_ip ON attack_event(client_ip);

-- 缓存刷新/预热任务
CREATE TABLE purge_task (
  id          bigserial PRIMARY KEY,
  domain_id   bigint NOT NULL REFERENCES domain(id) ON DELETE CASCADE,
  type        varchar(8) NOT NULL,                -- purge|prefetch
  targets     jsonb NOT NULL,                     -- url / 目录 / 全部
  status      varchar(16) NOT NULL DEFAULT 'queued',
  created_at  timestamptz NOT NULL DEFAULT now()
);
```

CREATE TABLE announcement (

## 9. 索引与性能要点

- `traffic_stat` 是写入与查询热点:按 `bucket` 时间分区(月分区)或上 **TimescaleDB**;查询永远带 `granularity` + 时间范围。
- `attack_event` 量大:分区 + 按套餐保留期定时清理(Starter 7 天 / Business 30 天 / Enterprise 90 天+)。
- 多租户查询强制带 `tenant_id`;在 ORM 层(NestJS 拦截器)注入,杜绝越权。
- 证书私钥、API secret 不明文落库:用对称加密(KMS / `pgcrypto`)。
- 配置下发以 `config_version` 为不可变快照,支持灰度(部分节点先应用)与一键回滚。
